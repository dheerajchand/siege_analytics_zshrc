```bash
# Backup and sync

sync_zsh                    # Sync repositories
list_backups                # List all backups

# Core utilities
modules                     # Show loaded modules
help                       # Show all commands
is_online                   # Check connectivity

# Python management  
py_env_switch list          # List Python environments
```
